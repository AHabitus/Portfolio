with rn as
    (select  *
             , row_number () over (partition by user_id order by date_purchase) as rn_user
    from cinema.client_sign_up
    )
select  count (*) cnt
        , is_trial
        , name_partner
        , rn_user
from rn as r
    left join cinema.partner_dict p
        on r.partner = p.id_partner
group by is_trial, name_partner, rn_user
order by name_partner
--
select  count(distinct date_purchase) as cnt_d_p
        , count(distinct case when t.rwn = 1 then date_purchase end) as cnt_rwn
        , count(distinct case when t.rnk = 1 then date_purchase end) as cnt_rnk
from (select date_purchase
             , row_number () over (order by date_purchase) as rwn
             , rank () over (order by date_purchase) as rnk
      from cinema.client_sign_up
     ) t
-----
select  id_client
        , name_city
        , case when gender = 'M' then 1 else 0 end nflag_gender
        , age
        , first_time
        , case when cellphone is not null then 1 else 0 end nflag_cellphone
        , is_active
        , cl_segm
        , amt_loan
        , date_loan :: date as date_loan
        , credit_type
        , sum (amt_loan) over (partition by name_city) sum_al_city
        , sum (amt_loan) :: float / sum (amt_loan) over (partition by name_city) share_city
        , sum (amt_loan) over (partition by credit_type) sum_al_type
        , sum (amt_loan) :: float / sum (amt_loan) over (partition by credit_type) share_type
        , sum (amt_loan) over (partition by name_city, credit_type) sum_city_type
        , sum (amt_loan)::float / sum(amt_loan) over (partition by name_city, credit_type) as share_city_type
        , count (amt_loan) over (partition by name_city) cnt_al_city
        , count (amt_loan) over (partition by credit_type) cnt_type
        , count (amt_loan) over (partition by name_city, credit_type) cnt_city_type
from bank.late_collection_clients a
    left join bank.region_dict b
        on a.id_city = b.id_city
group by id_client
         , name_city
         , nflag_gender
         , age
         , first_time
         , nflag_cellphone
         , is_active
         , cl_segm
         , amt_loan
         , date_loan
         , credit_type
